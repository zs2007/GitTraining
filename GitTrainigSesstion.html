Extended Git Command Reference Table
# Git command Notes
1 mkdir myfolder Create a new directory (OS-level, not tracked by Git unless it contains files)
2 touch newfile.txt Create a new file (Linux/Mac/Git Bash)
3 echo "Hello" > hello.txt Create a file with content
4 git add newfile.txt Add the newly created file to staging
5 git commit -m "Add new file" Commit the staged new file
6 echo "New content" > file.txt Replace contents of a file
7 git add file.txt Stage the modified file
8 git commit -m "Update file content" Commit the modified file
9 git mv oldname.txt newname.txt Rename a file
10 git mv file.txt folder/file.txt Move file to another directory
11 git commit -m "Rename file or move to folder" Commit the rename/move action
12 git rm file.txt Delete a file and stage the deletion
13 git rm -r folder/ Delete a directory and its contents
14 git commit -m "Delete file or folder" Commit the deletion
15 git clean -n Preview untracked files to be removed
16 git clean -f Remove untracked files from working directory
17 git checkout -- file.txt Discard changes in a specific file
18 git restore file.txt Restore file to last committed state
19 git diff file.txt Show unstaged changes in the file
20 git reset file.txt Unstage a file that was previously added
21 git grep "text" Search for text in tracked files
22 git grep -i "text" Case-insensitive search
23 git grep -l "text" Show filenames containing the text
24 git grep -il "text" Case-insensitive filenames only
25 git log --grep="keyword" Search commit messages by keyword
26 git log --author="name" Filter commit history by author
27 git log -S"text" Find commits that add/remove specific string
28 git blame file.txt Show who changed each line and when
29 git log --oneline Show concise commit history
30 git log --graph Display commit history as a graph
31 git log --since="2 weeks ago" Show commits within a date range
32 git log -p Show diffs introduced in each commit
33 git log -G"pattern" Search commits where code matches regex pattern
34 git blame Show who last modified each line in a file
35 git show Display details and changes of a specific commit
36 git show Show file content from the latest commit
37 git diff Compare changes between two commits
38 git ls-files List all tracked files
39 git ls-files | grep "pattern" Filter tracked files using a search pattern
40 git rev-list --all | xargs git grep "text" Search for text across all commits (deep search)
